-module(file_util).
%% Get file's line function
get_file_lines(FileData, Acc) ->
	case io:get_line(FileData, "") of
		eof -> 
			file:close(FileData),
			lists:reverse(Acc);
		Line -> get_file_lines(FileData, [Line|Acc])
	end.
